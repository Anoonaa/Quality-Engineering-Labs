This playground provides real-world UI patterns for practicing end-to-end testing with Playwright. Each page focuses on a different testing scenario — from form validation and API mocking to authentication flows and dynamic content rendering.

example.spec.ts 
import { test, expect } from '@playwright/test';

test('counter increments', async ({ page }) => {
  await page.goto('/');
  await page.getByTestId('increment-btn').click();
  await expect(page.getByTestId('counter-value'))
    .toHaveText('1');
});
